3. изучите с помощью функции var_dump(); следующие выражения:
    1. 3 / 1 // если оба числа являются целыми числами и делятся нацело,
             // то значением выражения то же будет целое число.
             // выведет int 3
    2. 1 / 3 // если числа не делятся нацело, то значением выражения будет число с плавающей точкой.
             // выведет float 0.33333333333333
    3. '20cats' + 40 // знак сложения приведет строку к числу.
                     // часть строки, которую нельзя привести к числу, будет отброшена.
                     // выведет int 60
    4.  18 % 4 // целочисленный остаток от деления. 4 * 4 = 16 и 2 остаток.
               // выведет int 2

4. что является значением выражения присваивания?
    1. echo ($a = 2); // значением выражения присваивания $a = 2 является 2
                      // будет выведено 2
    2. $x = ($y = 12) - 8; echo $x; // значением выражения присваивания $y = 12 является 12
                                    // значением выражения присваивания $x = 12 - 8 является 4
                                    // будет выведено 4

5. изучите с помощью var_dump(); следующие выражения и объясните их:
    1. 1 == 1.0 // первое число приведется к float. выведет true
    2. 1 === 1.0 // при строгом сравнении сравниваются еще и типы. так как типы разные выведет false
    3. '02' == 2 // строка приведется к числу 2. выведет true
    4. '02' === 2 // разные типы. выведет false
    5. '02' == '2' // при сравнении строк содержащих числа, каждая строка приводится к числу.
                   // выведет true
